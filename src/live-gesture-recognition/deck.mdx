### In the browser

```jsx
// @step showLines(12,28) highlight(28) name("Retrieve camera feed")
// @step showLines(13-14,16,24,18-20) highlight(19) name("Detect")
// @step showLines(15,17,21-23) highlight(21) name("Every 0.5 seconds")
// @step showLines(5-9,29) highlight(29) name("Display emoji")
import React, { useRef, useState, useEffect } from "react";
import Webcam from "react-webcam";
import { useCurrentGesture } from "./utils";

// Emoji mapping for recognized gestures
const SignToEmoji = {
  thumbs_up: "👍",
  // ...
};

function GestureRecognition() {
  const webcamRef = useRef(null);
  const { detect, gesture } = useCurrentGesture();

  // Run prediction every 500ms
  useEffect(() => {
    const interval = setInterval(() => {
      if (webcamRef.current?.video) {
        detect(webcamRef.current.video);
      }
    }, 500);

    return () => clearInterval(interval);
  }, [detect]);

  return (
    <div className="App">
      <Webcam ref={webcamRef} mirrored />
      <div className="emoji">{gesture ? SignToEmoji[gesture] : ""}</div>
    </div>
  );
}
```

### **Just more code**

```tsx
// @step showLines(10-12) highlight(10-11) name("Detect landmarks")
// @step showLines(13-17) highlight(13-17) name("Take only first one")
// @step showLines(18-21) highlight(18-19) name("Predict")
import * as tf from "@tensorflow/tfjs";
import { HandLandmarker } from "@mediapipe/tasks-vision";
import { useEffect, useState, useCallback } from "react";

export function useCurrentGesture() {
  const [gesture, setGesture] = useState(null);

  const detect = useCallback(
    async (video) => {
      const handLandmarks = handLandmarker.detect(video);
      if (!handLandmarks.landmarks?.length) return;

      const landmarks = handLandmarks.landmarks[0].flatMap((lm) => [
        lm.x,
        lm.y,
        lm.z,
      ]);
      const prediction = await model.predict(tf.tensor([landmarks])).array();
      const bestMatch = prediction[0].indexOf(Math.max(...prediction[0]));

      setGesture(gestureMap[bestMatch]);
    },
    [handLandmarker, model],
  );

  return { gesture, detect };
}
```
