### **How a typical React test looks**

```jsx
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";
import SurveyQuestion from "./SurveyQuestion";

describe("RatingSection", () => {
  it("allows user to select a rating", async () => {
    const mockQuestion = /*... */;
    const onSave = jest.fn();

    render(<SurveyQuestion question={mockQuestion} onSave={onSave} />);

    expect(screen.getByText("How was your experience?")).toBeVisible();
    expect(screen.getByText("Please rate from 1-5")).toBeVisible();

    const excellentButton = screen.getByRole("button", { name: "Excellent" });

    await userEvent.click(excellentButton);

    expect(onSave).toHaveBeenCalledWith(5);
  });
});
```

</Side>

### **The Three Components of POM**

<Side>

1. **Widget Objects**: Abstract UI elements and interactions
2. **Test Setup**: Component rendering with default props

### **Widget Object Example**

<Appear>

Kent will not be happy with this example 😅 (we are not using `screen`).

</Appear>

```tsx
// @step highlight(1-11) name("Widget Object")
// @step showLines(13-22) highlight(13-22) name("Elements")
// @step showLines(23-31) highlight(23-31) name("Actions")
export function RatingSectionWidgetObject({
  wrapper,
  user,
  content,
}: {
  wrapper?: HTMLElement;
  user: UserEvent;
  content: RatingSectionContent;
}) {
  const element = wrapper ?? window.document.body;

  return {
    elements: {
      title: () => getByText(element, content.text),
      subtext: () => getByText(element, content.subtext),
      ratingOption: (value: number) => {
        return screen.getByRole("button", {
          name: content.options.find((o) => o.value === value)
            ?.label,
        }),
      }
    },
    actions: {
      selectRating: async (value: number) => {
        const option = screen.getByRole("button", {
          name: content.options.find((o) => o.value === value)
            ?.label,
        });
        await user.click(option);
      },
    },
  };
}
```

### **Test Setup Example**

```tsx
// @step highlight(2) name("Overridable Props")
// @step showLines(4-14) highlight(4-14) name("Default Props")
// @step showLines(15-16) highlight(16) name("Render Component")
// @step showLines(17-27) highlight(18-27) name("Return component with widget object")
function renderComponent(
  props: Partial<Parameters<typeof RatingSection>[0]> = {},
) {
  const user = userEvent.setup();
  const onSave = vi.fn();
  const onDismiss = vi.fn();

  const defaultProps = {
    section: mockSection,
    currentStep: 1,
    totalSteps: 5,
    onSave,
    onDismiss,
  };

  const rendered = render(<RatingSection {...defaultProps} {...props} />);

  return {
    ...rendered,
    user,
    props: { ...defaultProps, ...props },
    ...RatingSectionWidgetObject({
      user,
      wrapper: rendered.container,
      content: mockSection.content,
    }),
  };
}
```

### **Test Cases Example**

```tsx
describe("RatingSection", () => {
  it("renders rating question with all options", () => {
    const { elements } = renderComponent();

    expect(elements.title()).toBeVisible();
    expect(elements.subtext()).toBeVisible();

    for (const option of mockSection.content.options) {
      expect(elements.ratingOption(option.value)).toBeVisible();
    }
  });

  it("calls onSave with selected rating value", async () => {
    const { actions, props } = renderComponent();

    await actions.selectRating(4);

    expect(props.onSave).toHaveBeenCalledWith(4);
  });

  it("returns null if question type is not RATING", () => {
    const { container } = renderComponent({
      section: survey.sections[0],
    });

    expect(container).toBeEmptyDOMElement();
  });
});
```

### **Bonus 1: Widget Objects are composable**

```tsx
function renderSurvey() {
  const user = userEvent.setup();

  const rendered = render(<NpsSurvey />);

  return {
    ...rendered,
    user,
    wait: {
      forSurveyToBeVisible: async () => {
        await waitFor(() => {
          expect(rendered.getByLabelText("NPS Survey")).toBeVisible();
        });
      },
    },
    actions: {
      openSurvey: () => user.click(rendered.getByText("Open Survey")),
    },
    widgets: {
      opening: OpeningSectionWidgetObject({
        user,
        wrapper: rendered.container,
      }),
      rating: RatingSectionWidgetObject({ user, wrapper: rendered.container }),
      choice: ChoiceSectionWidgetObject({
        user,
        wrapper: rendered.container,
      }),
      text: TextSectionWidgetObject({ user, wrapper: rendered.container }),
    },
  };
}
```

### **Bonus 1: Example**

```tsx
describe("Survey Flow", () => {
  it("completes full survey flow with positive feedback", async () => {
    const { widgets, actions, wait } = renderSurvey();

    await actions.openSurvey();
    await wait.forSurveyToBeVisible();

    // Step 1: Opening question
    await widgets.opening.actions.clickPrimary();

    // Step 2: Rating question
    await screen.findByText(
      "How do you describe your experience with this tool?",
    );
    await widgets.rating.actions.selectRating(5);

    // Step 3: Multiple choice
    await screen.findByText(
      "We're glad to hear that! What do you like most about the app?",
    );
    await widgets.multipleChoice.actions.selectOption("Usability");
    await widgets.multipleChoice.actions.enterFreeText(
      "Great user experience!",
    );
    await widgets.multipleChoice.actions.clickSave();

    // ...
  });
});
```

## **Benefits of POM Approach**

<Side>

</Appear>

<Appear>

- **Readability**: Tests focus on business scenarios, not DOM details

</Appear>

<Appear>

- **Scalability**: Complex flows become manageable

</Appear>

<Appear>

- **Resilience**: Less brittle tests that don't break with minor UI changes
